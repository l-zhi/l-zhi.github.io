js:动画通过css实现要更高效，通过z-index来控制遮罩修正侧滑时候的点击效果。

侧滑菜单的区域滚动实现方式：

var _drag = document.querySelector(‘.container’); var _H = -1 * $(‘.container’).height() + $(“#J_menuCotent”).height(); var _clientX,_clientY,rAF, target; var _touchstart = function(evt){ target = evt.targetTouches[0]; _clientY = target.pageY - (_drag._y || 0); $(‘.container’).css({‘-webkit-transform’: ‘translate3D(0px, 0, 0)’}); _drag.addEventListener(“touchmove”,update,true); _drag.addEventListener(“touchend”,clearEvent,true); draw(); //鼠标点击开始 evt.preventDefault(); } var clearEvent = function(evt){ _drag.removeEventListener(“touchmove”,update,true); _drag.removeEventListener(“touchend”,clearEvent,true); cancelAnimationFrame(rAF); } var update = function(evt){ target = evt.targetTouches[0]; _drag._y = target.pageY - _clientY; if(_drag._y > 0) _drag._y = 0; if(_drag._y < _H) _drag._y = _H; _drag._shouldMoved = true; } var draw = function(){ //没位移就不渲染 if (!_drag._shouldMoved) { rAF = requestAnimationFrame(draw); return; } _drag._lastY = _drag._y; _drag.style.webkitTransform = “translate3D(0px,” + _drag._y + “px,0px)”; _drag._shouldMoved = false; rAF = requestAnimationFrame(draw); } //RAF 支持~ requestAnimationFrame = window.requestAnimationFrame || window.mozRequestAnimationFrame || window.webkitRequestAnimationFrame || window.msRequestAnimationFrame || window.oRequestAnimationFrame || function(callback) {return setTimeout(callback, 1000 / 60); }; cancelAnimationFrame = window.cancelAnimationFrame || window.mozCancelAnimationFrame || window.webkitCancelAnimationFrame || window.msCancelAnimationFrame || window.oCancelAnimationFrame || function(callback) {return clearTimeout(callback, 1000 / 60); }; _drag.addEventListener(“touchstart”,_touchstart,true);

  • 通过requestAnimationFrame 来保证每一帧的效率。
  • 通过取消事件的绑定来释放不必要的内存，要知道移动端的内存很少。
  • 公用变量尽量提出来。
